/**
 * Definition for a Node.
 * type Node struct {
 *     Val int
 *     Left *Node
 *     Right *Node
 *     Next *Node
 * }
 */

func connect(root *Node) *Node {
	//终结者
    if root == nil {
        return root
    }
    //当前层逻辑
    root.Next = nil
    //下一层
    addNext(root.Left, root.Right)
    addNext(root.Right, nil)
    //返回
    return root
}

func addNext(root *Node, next *Node) {
    if root != nil {
        root.Next = next
        addNext(root.Left, root.Right)
        if root.Next == nil {
            addNext(root.Right,nil)
        } else {
            addNext(root.Right,root.Next.Left)
        }
        
    }
    
}

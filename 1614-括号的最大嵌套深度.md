// 记录（ 的深度，当）没有完全抵消（的时候则不在增加深度
func maxDepth(s string) int {
    var max int
    var tag int
    for _ , v := range s {
        if v == '(' && tag >= 0 {
            max += 1
        } else if v == ')' {
            tag -= 1
        } else if v == '(' {
            tag += 1
        }
    }
    return max
}
